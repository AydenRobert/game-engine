340KINLINE mat4 mat4_orthographic(f32 left, f32 right, f32 bottom, f32 top,
341 f32 near_clip, f32 far_clip) {
342 mat4 out_matrix = mat4_identity();
343
344 f32 lr = 1.0f / (left - right);
345 f32 bt = 1.0f / (bottom - top);
346 f32 nf = 1.0f / (near_clip - far_clip);
347
348 out_matrix.data[0] = -2.0f * lr;
349 out_matrix.data[5] = -2.0f * bt;
350 out_matrix.data[10] = -2.0f * nf;
351
352 out_matrix.data[12] = (left + right) * lr;
353 out_matrix.data[13] = (top + bottom) * bt;
354 out_matrix.data[14] = (far_clip + near_clip) * nf;
355
356 return out_matrix;
357}

359KINLINE mat4 mat4_perspective(f32 fov_radians, f32 aspect_ratio, f32 near_clip,
360 f32 far_clip) {

362 mat4 out_matrix;

373KINLINE mat4 mat4_look_at(vec3 position, vec3 target, vec3 up) {
374 mat4 out_matrix;
375 vec3 z_axis;
376 z_axis.x = target.x - position.x;
377 z_axis.y = target.y - position.y;
378 z_axis.z = target.z - position.z;

381 vec3 x_axis = vec3_normalized(vec3_cross(z_axis, up));
382 vec3 y_axis = vec3_cross(x_axis, z_axis);

384 out_matrix.data[0] = x_axis.x;
385 out_matrix.data[1] = y_axis.x;
386 out_matrix.data[2] = -z_axis.x;
387 out_matrix.data[3] = 0;
388 out_matrix.data[4] = x_axis.y;
389 out_matrix.data[5] = y_axis.y;
390 out_matrix.data[6] = -z_axis.y;
391 out_matrix.data[7] = 0;
392 out_matrix.data[8] = x_axis.z;
393 out_matrix.data[9] = y_axis.z;
394 out_matrix.data[10] = -z_axis.y;

396 out_matrix.data[12] = -vec3_dot(x_axis, position);
397 out_matrix.data[13] = -vec3_dot(y_axis, position);
398 out_matrix.data[14] = vec3_dot(z_axis, position);
399 out_matrix.data[15] = 1.0f;

658KINLINE mat4 quat_to_mat4(quat q) {
659 mat4 out_matrix = mat4_identity();
660
661 // https://stackoverflow.com/questions/1556260/convert-quaternion-rotation-to-rotation-matrix
662
663 quat n = quat_normalize(q);
664
665 out_matrix.data[0] = 1.0f - 2.0f * n.y * n.y - 2.0f * n.z * n.z;
666 out_matrix.data[1] = 2.0f * n.x * n.y - 2.0f * n.z * n.w;
667 out_matrix.data[2] = 2.0f * n.x * n.z + 2.0f * n.y * n.w;
668
669 out_matrix.data[4] = 2.0f * n.x * n.y + 2.0f * n.z * n.w;
670 out_matrix.data[5] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.z * n.z;
671 out_matrix.data[6] = 2.0f * n.y * n.z - 2.0f * n.x * n.w;
672
673 out_matrix.data[8] = 2.0f * n.x * n.z - 2.0f * n.y * n.w;
674 out_matrix.data[9] = 2.0f * n.y * n.z + 2.0f * n.x * n.w;
675 out_matrix.data[10] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.y * n.y;
676
677 return out_matrix;
678}

680KINLINE mat4 quat_to_rotation_matrix(quat q, vec3 center) {
681 mat4 out_matrix;
682
683 f32 *o = out_matrix.data;
684 o[0] = (q.x * q.x) - (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
685 o[1] = 2.0f * ((q.x * q.y) + (q.z * q.w));
686 o[2] = 2.0f * ((q.x * q.z) - (q.y * q.w));
687 o[3] = center.x - center.x * o[0] - center.y * o[1] - center.z * o[2];
688
689 o[4] = 2.0f * ((q.x * q.y) - (q.z * q.w));
690 o[5] = -(q.x * q.x) + (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
691 o[6] = 2.0f * ((q.y * q.z) + (q.x * q.w));
692 o[7] = center.y - center.x * o[4] - center.y * o[5] - center.z * o[6];
693
694 o[8] = 2.0f * ((q.x * q.z) + (q.y * q.w));
695 o[9] = 2.0f * ((q.y * q.z) - (q.x * q.w));
696 o[10] = -(q.x * q.x) - (q.y * q.y) + (q.z * q.z) + (q.w * q.w);
697 o[11] = center.z - center.x * o[8] - center.y * o[9] - center.z * o[10];
698
699 o[12] = 0.0f;
700 o[13] = 0.0f;
701 o[14] = 0.0f;
702 o[15] = 1.0f;
703 return out_matrix;
704}

718KINLINE quat quat_slerp(quat q_0, quat q_1, f32 percentage) {
719
720 quat out_quaternion;
721 // Source: https://en.wikipedia.org/wiki/Slerp
722 // Only unit quaternions are valid rotations.
723 // Normalize to avoid undefined behavior.
724 quat v0 = quat_normalize(q_0);
725 quat v1 = quat_normalize(q_1);
726
727 // Compute the cosine of the angle between the two vectors.
728 f32 dot = quat_dot(v0, v1);
729
730 // If the dot product is negative, slerp won't take
731 // the shorter path. Note that v1 and -v1 are equivalent when
732 // the negation is applied to all four components. Fix by
733 // reversing one quaternion.
734 if (dot < 0.0f) {
735 v1.x = -v1.x;
736 v1.y = -v1.y;
737 v1.z = -v1.z;
738 v1.w = -v1.w;
739 dot = -dot;
740 }
741
742 const f32 DOT_THRESHOLD = 0.9995f;
743 if (dot > DOT_THRESHOLD) {
744 // If the inputs are too close for comfort, linearly interpolate
745 // and normalize the result.
746 out_quaternion = (quat){{v0.x + ((v1.x - v0.x) * percentage),
747 v0.y + ((v1.y - v0.y) * percentage),
748 v0.z + ((v1.z - v0.z) * percentage),
749 v0.w + ((v1.w - v0.w) * percentage)}};
750
751 return quat_normalize(out_quaternion);
752 }
753
754 // Since dot is in range [0, DOT_THRESHOLD], acos is safe
755 f32 theta_0 = kacos(dot); // theta_0 = angle between input vectors
756 f32 theta = theta_0 * percentage; // theta = angle between v0 and result
757 f32 sin_theta = ksin(theta); // compute this value only once
758 f32 sin_theta_0 = ksin(theta_0); // compute this value only once
759
760 f32 s0 =
761 kcos(theta) -
762 dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
763 f32 s1 = sin_theta / sin_theta_0;
764
765 return (quat){{(v0.x * s0) + (v1.x * s1), (v0.y * s0) + (v1.y * s1),
766 (v0.z * s0) + (v1.z * s1), (v0.w * s0) + (v1.w * s1)}};
767}
